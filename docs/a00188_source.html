106 volatile uint16_t appLCD_Execute(void)
107 {
108  volatile uint16_t retval = 1;
109  volatile float vi=0.0, vo=0.0, isns=0.0, temp=0.0;
110 
111  // IF LCD output is disabled, exit here
112  if(!lcd.enabled)
113  return(retval);
114 
115  // Refresh LCD and reset refresh counter
116  lcd_cnt++;
117 
118  // If REFRESH period has expired, update LCD contents
119  if(lcd_cnt == lcd.refresh) {
120 
121  // Calculate output values
122  vi = ((boost.data.v_in << 3) * ADC_GRANULARITY); // Scale ADC value to physical unit
123  vi = (float)(int)(100.0 * vi); // Rounding operation required to prevent display
124  vi /= 100.0; // rounding issues around 9.99 and 10.0 ° C
125 
126  // Input voltage display
127  if((double)vi < 10.000)
128  PrintLcd(0, "VIN = %2.2f V", (double)vi);
129  else
130  PrintLcd(0, "VIN = %2.1f V", (double)vi);
131 
132  switch (lcd.screen)
133  {
134  case 1: // Show Temperature Output
135 
136  temp = ((float)(boost.data.temp - BOOST_FB_TEMP_ZERO) / BOOST_FB_TEMP_SLOPE); // Scale ADC value to physical unit
137  temp = (float)(int)(100.0 * temp); // Rounding operation required to prevent display
138  temp /= 100.0; // rounding issues around 9.99 and 10.0 V
139 
140  if((double)temp < 10.000)
141  PrintLcd(1, "TEMP = %2.2f C", (double)temp);
142  else
143  PrintLcd(1, "TEMP = %2.1f C", (double)temp);
144  break;
145 
146  case 2: // Show Current Output
147 
148  isns = ((boost.data.i_out * ADC_GRANULARITY) / BOOST_ISNS_FEEDBACK_GAIN); // Scale ADC value to physical unit
149 
150  if((double)isns < 1.000)
151  {
152  isns *= 1000.0;
153  PrintLcd(1, "ISNS = %3d mA", (int)isns);
154  }
155  else
156  {
157  PrintLcd(1, "ISNS = %1.2f A", (double)isns);
158  }
159  break;
160 
161  default: // Output voltage display
162 
163  vo = ((boost.data.v_out << 3) * ADC_GRANULARITY); // Scale ADC value to physical unit
164 
165  if((double)vo < 10.000)
166  PrintLcd(1, "VOUT = %2.2f V", (double)vo);
167  else
168  PrintLcd(1, "VOUT = %2.1f V", (double)vo);
169 
170  if (boost.status.bits.fault_active)
171  {
172  if (fltobj_BoostUVLO.Status.bits.FaultStatus)
173  dev_Lcd_WriteStringXY(4, 1, "(UV)");
174  else if (fltobj_BoostOVLO.Status.bits.FaultStatus)
175  dev_Lcd_WriteStringXY(4, 1, "(OV)");
176  else if (fltobj_BoostRegErr.Status.bits.FaultStatus)
177  dev_Lcd_WriteStringXY(4, 1, "(RE)");
178  }
179  break;
180  }
181 
182  lcd.refresh = LCD_REFRESH;
183 
184  lcd_cnt = 0; // Reset internal interval counter
185  }
186 
187  return(retval);
188 }
