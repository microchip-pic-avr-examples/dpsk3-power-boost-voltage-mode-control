79 volatile uint16_t appFaultMonitor_Execute(void)
80 {
81  volatile uint16_t retval=1;
82  static bool pre_fault_active=false;
83 
84  // Call fault handler
85  retval &= drv_FaultHandler_CheckObject(&fltobj_BoostUVLO);
86  retval &= drv_FaultHandler_CheckObject(&fltobj_BoostOVLO);
87  retval &= drv_FaultHandler_CheckObject(&fltobj_BoostRegErr);
88  retval &= drv_FaultHandler_CheckObject(&fltobj_BoostOCP);
89 
90  // Combine individual fault bits to a common fault indicator
91  boost.status.bits.fault_active = (bool) (
92  fltobj_BoostUVLO.Status.bits.FaultStatus |
93  fltobj_BoostOVLO.Status.bits.FaultStatus |
94  fltobj_BoostRegErr.Status.bits.FaultStatus |
95  fltobj_BoostOCP.Status.bits.FaultStatus
96  );
97 
98  // If system has recovered from a global fault condition,
99  // trigger fault recovery function
100  if ((pre_fault_active) && (!boost.status.bits.fault_active))
101  { retval = appPowerSupply_Resume(); }
102 
103  // Track global fault bit transitions
104  pre_fault_active = boost.status.bits.fault_active;
105 
106 
107  return (retval);
108 }

179 volatile uint16_t uvlo_FaultInitialize(void)
180 {
181  volatile uint16_t retval=1;
182 
183  // Initialize UVLO fault object
184  fltobj_BoostUVLO = fltObjectClear; // Pre-initialize fault object
185 
186  fltobj_BoostUVLO.SourceObject.ptrObject = &boost.data.v_in; // Set pointer to variable to monitor
187  fltobj_BoostUVLO.SourceObject.bitMask = 0xFFFF; // Compare all bits of SOURCE (no bit filter)
188  fltobj_BoostUVLO.ReferenceObject.ptrObject = NULL; // Clear pointer to "compare against" variable
189  fltobj_BoostUVLO.ReferenceObject.bitMask = 0xFFFF; // Clear pointer to "compare against" variable
190 
191  fltobj_BoostUVLO.Status.bits.CompareType = FLTCMP_LESS_THAN; // Select Compare-Type
192 
193  fltobj_BoostUVLO.TripResponse.compareThreshold = BOOST_VIN_UVLO_TRIP; // Set fault trip level
194  fltobj_BoostUVLO.TripResponse.eventThreshold = BOOST_UVLO_TDLY; // Set counter level at which a FAULT condition will be tripped
195  fltobj_BoostUVLO.TripResponse.ptrResponseFunction = &appPowerSupply_Suspend; // Set pointer to user-function which should be called when a FAULT is tripped
196 
197  fltobj_BoostUVLO.RecoveryResponse.compareThreshold = BOOST_VIN_UVLO_RELEASE; // Set fault recovery level
198  fltobj_BoostUVLO.RecoveryResponse.eventThreshold = BOOST_UVLO_RDLY; // Set counter level at which a FAULT condition will be cleared
199  fltobj_BoostUVLO.RecoveryResponse.ptrResponseFunction = NULL; // Clear recovery function pointer
200 
201  fltobj_BoostUVLO.Counter = 0; // Clear fault event counter
202  fltobj_BoostUVLO.Status.bits.FaultActive = true; // Set fault condition flag (must be cleared by fault check)
203  fltobj_BoostUVLO.Status.bits.FaultStatus = true; // Set fault flag (must be cleared by fault check)
204  fltobj_BoostUVLO.Status.bits.Enabled = true; // Enable fault checks
205 
206  return(retval);
207 
208 }
