55 volatile uint16_t Substate_VCapMonitor(volatile struct BOOST_CONVERTER_s *boostInstance)
56 {
57  volatile uint16_t _source=0;
58 
59  // Set BUSY bit until process is complete
60  boostInstance->status.bits.busy = true;
61 
62  // Capture compare value
63  _source = abs(boostInstance->data.v_in - boostInstance->data.v_out);
64 
65  if((_source > boostInstance->startup.vcap_monitor.v_drop) &&
66  (boostInstance->startup.vcap_monitor.period > 0))
67  boostInstance->startup.vcap_monitor.counter = 0; // Reset counter if voltage is not tuned in yet
68 
69  // Monitor settled voltage for the given period of time
70  if (++boostInstance->startup.vcap_monitor.counter >
71  boostInstance->startup.vcap_monitor.period)
72  {
73  boostInstance->startup.vcap_monitor.counter = 0;
74  boostInstance->startup.vcap_monitor.timeout_counter = 0;
75 
76  Nop(); // Please Debugging Breakpoint
77  Nop();
78  Nop();
79 
80  return(BOOST_OPSRET_COMPLETE);
81 
82  }
83 
84  // In case of a timeout, reset state machine and try again
85  if (boostInstance->startup.vcap_monitor.timeout_counter++ >
86  boostInstance->startup.vcap_monitor.timeout)
87  {
88  boostInstance->startup.vcap_monitor.counter = 0;
89  boostInstance->startup.vcap_monitor.timeout_counter = 0;
90 
91  Nop(); // Please Debugging Breakpoint
92  Nop();
93  Nop();
94 
95  return (BOOST_OPSRET_ERROR);
96 
97  }
98 
99  // Recall this sub-state
100  return(BOOST_OPSRET_REPEAT);
101 
102 }

133  else

162 volatile uint16_t SubState_PrepareVRampUp(volatile struct BOOST_CONVERTER_s *boostInstance)
163 {
164  volatile uint16_t _i=0;
165  volatile uint32_t _vout=0;
166  volatile uint32_t _vin=0;
167  volatile uint32_t _start_dc=0;
168 
169  // Set BUSY bit until process is complete
170  boostInstance->status.bits.busy = true;
171 
172  // Hijack voltage loop controller reference
173  boostInstance->startup.v_ramp.reference = 0; // Reset Soft-Start Voltage Reference
174  boostInstance->startup.i_ramp.reference = 0; // Reset Soft-Start Current Reference
175  boostInstance->v_loop.controller->Ports.ptrControlReference =
176  &boostInstance->startup.v_ramp.reference; // Voltage loop is pointing to Soft-Start Reference
177 
178  // Pre-charge reference
179  // Never start above the pre-biased output voltage.
180  // Always start at or slightly below the pre-biased output voltage
181  boostInstance->startup.v_ramp.reference = boostInstance->data.v_out;
182 
183  // In average current mode, set current reference limit to max startup current level
184  if (boostInstance->set_values.control_mode == BOOST_CONTROL_MODE_ACMC)
185  { // Disable all current control loops and reset control loop histories
186  boostInstance->v_loop.maximum = boostInstance->set_values.i_ref_startup;
187  boostInstance->v_loop.controller->Limits.MaxOutput = boostInstance->v_loop.maximum;
188  }
189 
190  // Pre-charge PWM and control loop history
191  if(((boostInstance->data.v_in - boostInstance->feedback.ad_vin.scaling.offset) > 0) &&
192  ((boostInstance->data.v_out - boostInstance->feedback.ad_vout.scaling.offset) > 0) )
193  {
194  _vout = __builtin_muluu(
195  (boostInstance->data.v_out - boostInstance->feedback.ad_vout.scaling.offset),
196  boostInstance->feedback.ad_vout.scaling.factor);
197  _vout >>= (16 - boostInstance->feedback.ad_vout.scaling.scaler);
198 
199  _vin = __builtin_muluu(
200  (boostInstance->data.v_in - boostInstance->feedback.ad_vin.scaling.offset),
201  boostInstance->feedback.ad_vin.scaling.factor);
202  _vin >>= (16 - boostInstance->feedback.ad_vin.scaling.scaler);
203 
204  // Protect against negative duty cycle results
205  if (_vout < _vin) _vout = _vin;
206 
207  // CALCULATE BOOST CONVERTER STARTUP DUTY RATIO
208  // DC = (VOUT-VIN) / VOUT, where DC = D * PERIOD
209 
210  if(_vin > 0)
211  {
212  _start_dc = __builtin_muluu((_vout-_vin), boostInstance->sw_node[0].period);
213  _start_dc = __builtin_divud(_start_dc, (uint16_t)_vout);
214  }
215  else
216  { _start_dc = (uint16_t)boostInstance->sw_node[_i].duty_ratio_min; }
217  }
218  else
219  // If there is no input voltage or no output voltage, start with minimum duty ratio
220  {
221  for (_i=0; _i<boostInstance->set_values.no_of_phases; _i++) {
222  _start_dc = (uint16_t)boostInstance->sw_node[_i].duty_ratio_min;
223  }
224  }
225 
226  // Program PWM module and pre-charge control loop
227  if (boostInstance->set_values.control_mode == BOOST_CONTROL_MODE_VMC)
228  {
229  if(_start_dc < boostInstance->v_loop.minimum)
230  { _start_dc = boostInstance->v_loop.minimum; }
231  else if(_start_dc > boostInstance->v_loop.maximum)
232  { _start_dc = boostInstance->v_loop.maximum; }
233  else { /* continue */ }
234 
235  boostInstance->v_loop.ctrl_Precharge(boostInstance->v_loop.controller, 0, _start_dc);
236  *boostInstance->v_loop.controller->Ports.Target.ptrAddress = _start_dc; // set initial PWM duty ratio
237  if(boostInstance->v_loop.controller->Ports.AltTarget.ptrAddress != NULL)
238  *boostInstance->v_loop.controller->Ports.AltTarget.ptrAddress = _start_dc; // set initial PWM duty ratio
239  else { /* continue */ }
240 
241  }
242  else if (boostInstance->set_values.control_mode == BOOST_CONTROL_MODE_ACMC)
243  {
244  // Limit startup duty cycle to largest minimum/smallest maximum
245  for (_i=0; _i<boostInstance->set_values.no_of_phases; _i++) {
246 
247  if(_start_dc < boostInstance->i_loop[_i].minimum)
248  { _start_dc = boostInstance->i_loop[_i].minimum; }
249  else if(_start_dc > boostInstance->i_loop[_i].maximum)
250  { _start_dc = boostInstance->i_loop[_i].maximum; }
251  else { /* continue */ }
252  }
253 
254  // Set initial duty cycle
255  for (_i=0; _i<boostInstance->set_values.no_of_phases; _i++)
256  {
257  // pre-charge current loop control histories with ideal duty cycle
258  boostInstance->i_loop[_i].ctrl_Precharge(
259  boostInstance->i_loop[_i].controller, 0, _start_dc
260  );
261 
262  // pre-charge PWM outputs with ideal duty cycle
263  *boostInstance->i_loop[_i].controller->Ports.Target.ptrAddress = _start_dc; // set initial PWM duty ratio
264  if(boostInstance->i_loop[_i].controller->Ports.AltTarget.ptrAddress != NULL)
265  *boostInstance->i_loop[_i].controller->Ports.AltTarget.ptrAddress = _start_dc; // set initial PWM duty ratio
266  else { /* continue */ }
267  }
268  }
269  else
270  {
271  return(BOOST_OPSRET_ERROR);
272  }
273 
274  return(BOOST_OPSRET_COMPLETE);
275 
276 }

294 volatile uint16_t SubState_VRampUp(volatile struct BOOST_CONVERTER_s *boostInstance)
295 {
296  volatile uint16_t retval=1;
297  volatile uint16_t _i=0;
298 
299  // Set BUSY bit until process is complete
300  boostInstance->status.bits.busy = true;
301 
302  // ensure control loop is enabled
303  if(!boostInstance->v_loop.controller->status.bits.enabled) {
304 
305  // Enable all PWM channels
306  retval &= boostPWM_Resume(boostInstance); // Enable PWM outputs
307 
308  // In voltage mode control...
309  if (boostInstance->set_values.control_mode == BOOST_CONTROL_MODE_VMC)
310  { // only enable voltage loop
311  boostInstance->v_loop.controller->status.bits.enabled = true; // enable voltage loop controller
312  }
313  // In average current mode control...
314  else if (boostInstance->set_values.control_mode == BOOST_CONTROL_MODE_ACMC)
315  {
316  // enable voltage loop controller
317  boostInstance->v_loop.controller->status.bits.enabled = true;
318 
319  // enable all phase current loop controllers
320  for (_i=0; _i<boostInstance->set_values.no_of_phases; _i++) {
321  boostInstance->i_loop[_i].controller->status.bits.enabled = true;
322  }
323  }
324  // IF control mode is set to unsupported control mode, return error
325  else
326  {
327  return(BOOST_OPSRET_ERROR);
328  }
329 
330  }
331 
332  // increment reference
333  boostInstance->startup.v_ramp.reference += boostInstance->startup.v_ramp.ref_inc_step;
334 
335  // check if ramp is complete
336  if (boostInstance->startup.v_ramp.reference > boostInstance->v_loop.reference)
337  {
338  // Set reference to the desired level
339  boostInstance->startup.v_ramp.reference = boostInstance->v_loop.reference;
340 
341  // Reconnect API reference to controller
342  boostInstance->v_loop.controller->Ports.ptrControlReference = &boostInstance->v_loop.reference;
343 
344  retval = BOOST_OPSRET_COMPLETE;
345 
346  }
347  else
348  // remain in this state until ramp is complete
349  {
350  retval = BOOST_OPSRET_REPEAT;
351  }
352 
353 
354  return(retval);
355 
356 }

377 volatile uint16_t SubState_IRampUp(volatile struct BOOST_CONVERTER_s *boostInstance)
378 {
379  volatile uint16_t retval=0;
380 
381  // Set BUSY bit until process is complete
382  boostInstance->status.bits.busy = true;
383 
384  // in average current mode if voltage limit is hit, increment
385  // voltage loop limit to ramp up current loop
386  if (boostInstance->set_values.control_mode == BOOST_CONTROL_MODE_ACMC)
387  {
388  // Increment maximum current limit
389  boostInstance->v_loop.controller->Limits.MaxOutput += boostInstance->startup.i_ramp.ref_inc_step;
390 
391  if (boostInstance->v_loop.controller->Limits.MaxOutput >= boostInstance->set_values.i_ref_startup)
392  // check if ramp is complete
393  {
394  boostInstance->v_loop.maximum = boostInstance->set_values.i_ref_startup;
395  boostInstance->v_loop.controller->Limits.MaxOutput = boostInstance->v_loop.maximum;
396  retval = BOOST_OPSRET_COMPLETE;
397  }
398  else
399  // if ramp is not complete yet, remain in this state
400  {
401  retval = BOOST_OPSRET_REPEAT;
402  }
403  }
404  else // Non-Current Loops Ending up here need to be lifted to PG_DELAY
405  {
406  boostInstance->v_loop.controller->Limits.MaxOutput = boostInstance->v_loop.maximum;
407  retval = BOOST_OPSRET_COMPLETE;
408  }
409 
410  return(retval);
411 
412 }

427 volatile uint16_t SubState_PowerGoodDelay(volatile struct BOOST_CONVERTER_s *boostInstance)
428 {
429  volatile uint16_t retval=0;
430 
431  // Set BUSY bit until process is complete
432  boostInstance->status.bits.busy = true;
433 
434  // increment delay counter until the GOWER GOOD delay has expired
435  if(boostInstance->startup.power_good_delay.counter++ > boostInstance->startup.power_good_delay.period)
436  {
437  boostInstance->startup.power_good_delay.counter =
438  (boostInstance->startup.power_good_delay.period + 1); // Clamp to PERIOD_EXPIRED for future startups
439 
440  // If defined, set POWER_GOOD output
441  if(boostInstance->gpio.PowerGood.enabled)
442  {
443  retval = boostGPIO_Set(&boostInstance->gpio.PowerGood);
444 
445  if(!retval)
446  { retval = BOOST_OPSTATE_ERROR; }
447  }
448 
449  retval = BOOST_OPSRET_COMPLETE;
450 
451  }
452  else
453  // if period has not expired yet, remain in this state
454  {
455  retval = BOOST_OPSRET_REPEAT;
456  }
457 
458  return(retval);
